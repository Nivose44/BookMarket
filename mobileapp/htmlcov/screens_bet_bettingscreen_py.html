38 def create_view(self, choice):

40 Depending of the current_page value, call the function creating the page

43 self.sport_chosen = choice if choice else self.sport_chosen

45 0: self.create_sport_page,

51 def create_sport_page(self):

53 Create the page to chose the sports to get bets from or chose between other
